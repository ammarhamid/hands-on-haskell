Pattern Matching on lists

Pattern match on [] and :
> and :: [Bool] -> Bool
> and []     = True
> and (x:xs) = x && and xs
      
> or :: [Bool] -> Bool
> or []     = False
> or (x:xs) = x || or xs

Generalizing list functions

> and [] = True
> or  [] = False
      
> and (x:xs) = x && and xs
> or  (x:xs) = x || or  xs
Combine function f + default case d:

Generalizing list functions

> foldr :: (a -> b -> b) -> b -> [a] -> b
> foldr _ d []     = d
> foldr f d (x:xs) = f x (foldr f d xs)

Map

> map _ []     = []
> map f (x:xs) = f x : map f xs
      
> map f = foldr (\a b -> f a : b) []

filter

> filter _ []     = []
> filter p (x:xs) =
>   if p x
>   then x : filter p xs
>   else filter p xs
      
> filter p = foldr step []
>   where step a b = if p a
>                    then a : b
>                    else b
